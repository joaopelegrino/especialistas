    def assess_asset_criticality(self, asset_data):
        """Assess asset criticality based on business impact"""
        criticality_score = 0

        # Business process dependency
        if asset_data.get('supports_critical_process'):
            criticality_score += 40

        # Data sensitivity
        sensitivity = asset_data.get('data_sensitivity', 'public')
        sensitivity_scores = {
            'restricted': 30,
            'confidential': 20,
            'internal': 10,
            'public': 0
        }
        criticality_score += sensitivity_scores.get(sensitivity, 0)

        # User dependency
        user_count = asset_data.get('user_count', 0)
        if user_count > 1000:
            criticality_score += 20
        elif user_count > 100:
            criticality_score += 10
        elif user_count > 10:
            criticality_score += 5

        # Revenue impact
        revenue_impact = asset_data.get('revenue_impact', 'none')
        revenue_scores = {
            'direct': 30,
            'indirect': 15,
            'support': 5,
            'none': 0
        }
        criticality_score += revenue_scores.get(revenue_impact, 0)

        # Determine criticality level
        if criticality_score >= 80:
            return 'critical'
        elif criticality_score >= 60:
            return 'high'
        elif criticality_score >= 40:
            return 'medium'
        else:
            return 'low'

#!/usr/bin/env python3
"""
NIST CSF Maturity Assessment Framework
Comprehensive organizational security maturity evaluation
"""

class NISTCSFAssessment:
    def __init__(self):
        self.csf_functions = {
            'identify': {
                'asset_management': ['ID.AM'],
                'business_environment': ['ID.BE'],
                'governance': ['ID.GV'],
                'risk_assessment': ['ID.RA'],
                'risk_management_strategy': ['ID.RM'],
                'supply_chain': ['ID.SC']
            },
            'protect': {
                'identity_management': ['PR.AC'],
                'awareness_training': ['PR.AT'],
                'data_security': ['PR.DS'],
                'information_protection': ['PR.IP'],
                'maintenance': ['PR.MA'],
                'protective_technology': ['PR.PT']
            },
            'detect': {
                'anomalies_events': ['DE.AE'],
                'security_monitoring': ['DE.CM'],
                'detection_processes': ['DE.DP']
            },
            'respond': {
                'response_planning': ['RS.RP'],
                'communications': ['RS.CO'],
                'analysis': ['RS.AN'],
                'mitigation': ['RS.MI'],
                'improvements': ['RS.IM']
            },
            'recover': {
                'recovery_planning': ['RC.RP'],
                'improvements': ['RC.IM'],
                'communications': ['RC.CO']
            }
        }

        self.maturity_levels = {
            1: 'Partial',
            2: 'Risk-Informed',
            3: 'Repeatable',
            4: 'Adaptive'
        }

    def assess_subcategory(self, subcategory, evidence):
        """Assess maturity level for specific subcategory"""
        scoring_criteria = {
            1: {  # Partial
                'description': 'Ad hoc processes, limited documentation',
                'indicators': ['informal_processes', 'limited_awareness', 'reactive_approach']
            },
            2: {  # Risk-Informed
                'description': 'Risk management practices approved and expressed as policy',
                'indicators': ['documented_processes', 'risk_awareness', 'some_automation']
            },
            3: {  # Repeatable
                'description': 'Organizational cybersecurity practices regularly updated',
                'indicators': ['standardized_processes', 'regular_updates', 'metrics_tracking']
            },
            4: {  # Adaptive
                'description': 'Adaptive and predictive cybersecurity practices',
                'indicators': ['continuous_improvement', 'predictive_capabilities', 'integrated_approach']
            }
        }

        # Score based on evidence provided
        score = self.calculate_score(evidence, scoring_criteria)
        return score

    def conduct_comprehensive_assessment(self, organization_data):
        """Conduct full NIST CSF maturity assessment"""
        assessment_results = {}

        for function_name, categories in self.csf_functions.items():
            function_scores = {}

            for category_name, subcategories in categories.items():
                category_scores = []

                for subcategory in subcategories:
                    evidence = organization_data.get(subcategory, {})
                    score = self.assess_subcategory(subcategory, evidence)
                    category_scores.append(score)

                # Average score for category
                function_scores[category_name] = sum(category_scores) / len(category_scores)

            # Average score for function
            assessment_results[function_name] = sum(function_scores.values()) / len(function_scores)

        return assessment_results

    def generate_maturity_report(self, assessment_results, target_levels):
        """Generate comprehensive maturity assessment report"""
        gaps = {}
        priorities = []

        for function, current_level in assessment_results.items():
            target = target_levels.get(function, 3)  # Default target level 3
            gap = target - current_level

            if gap > 0:
                gaps[function] = {
                    'current': current_level,
                    'target': target,
                    'gap': gap,
                    'priority': self.calculate_priority(function, gap)
                }

        # Sort by priority
        sorted_gaps = sorted(gaps.items(), key=lambda x: x[1]['priority'], reverse=True)

        report = {
            'executive_summary': self.create_executive_summary(assessment_results, gaps),
            'detailed_findings': assessment_results,
            'gap_analysis': gaps,
            'prioritized_roadmap': self.create_roadmap(sorted_gaps),
            'investment_recommendations': self.calculate_investment_needs(gaps)
        }

        return report

    def create_executive_summary(self, results, gaps):
        """Create executive summary for leadership"""
        overall_maturity = sum(results.values()) / len(results)

        summary = {
            'overall_maturity_level': round(overall_maturity, 1),
            'maturity_description': self.maturity_levels.get(round(overall_maturity), 'Between levels'),
            'key_strengths': self.identify_strengths(results),
            'critical_gaps': self.identify_critical_gaps(gaps),
            'business_risk_impact': self.assess_business_impact(gaps),
            'investment_summary': self.summarize_investment_needs(gaps)
        }

        return summary

    def create_roadmap(self, prioritized_gaps):
        """Create implementation roadmap"""
        roadmap = {
            'phase_1': {  # 0-6 months
                'duration': '0-6 months',
                'focus': 'Critical gaps and quick wins',
                'initiatives': []
            },
            'phase_2': {  # 6-12 months
                'duration': '6-12 months',
                'focus': 'Process improvement and automation',
                'initiatives': []
            },
            'phase_3': {  # 12-24 months
                'duration': '12-24 months',
                'focus': 'Advanced capabilities and optimization',
                'initiatives': []
            }
        }

        # Distribute initiatives across phases based on priority and complexity
        for function, gap_data in prioritized_gaps:
            initiative = self.create_initiative(function, gap_data)

            if gap_data['priority'] >= 8:
                roadmap['phase_1']['initiatives'].append(initiative)
            elif gap_data['priority'] >= 5:
                roadmap['phase_2']['initiatives'].append(initiative)
            else:
                roadmap['phase_3']['initiatives'].append(initiative)

        return roadmap

def calculate_annualized_loss_expectancy(asset_value, threat_frequency, vulnerability_probability, impact_percentage):
    """
    ALE = AV √ó EF √ó ARO
    Where:
    AV = Asset Value
    EF = Exposure Factor (impact percentage)
    ARO = Annual Rate of Occurrence (threat frequency √ó vulnerability probability)
    """
    exposure_factor = impact_percentage / 100
    annual_rate_occurrence = threat_frequency * vulnerability_probability

    single_loss_expectancy = asset_value * exposure_factor
    annualized_loss_expectancy = single_loss_expectancy * annual_rate_occurrence

    return {
        'sle': single_loss_expectancy,
        'ale': annualized_loss_expectancy,
        'aro': annual_rate_occurrence
    }

# Example calculation
result = calculate_annualized_loss_expectancy(
    asset_value=1000000,      # $1M system
    threat_frequency=0.1,     # 10% chance per year
    vulnerability_probability=0.3,  # 30% probability if threat occurs
    impact_percentage=25      # 25% impact if successful
)

print(f"Single Loss Expectancy: ${result['sle']:,.2f}")
print(f"Annualized Loss Expectancy: ${result['ale']:,.2f}")
